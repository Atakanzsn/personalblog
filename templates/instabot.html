                def instagram(self):
                    self.label_7.setText("program Çalışıyor, Lütfen Kapatmayınız veya Tıklamayınız.")
                    try:
                        self.usern=self.lineEdit.text()
                        self.passw=self.lineEdit_2.text()
                        self.tagss = self.lineEdit_6.text().split(",")
                        if self.lineEdit_7.text() == None:
                            self.likem = 5
                        else:
                            self.likem = self.lineEdit_7.text()
                        if self.radioButton.isChecked() == True:
                            self.follows = True
                        else:
                            self.follows = False
                            
                        if self.lineEdit_3.text() == None:
                            self.followp = 0
                        else:
                            self.followp = self.lineEdit_3.text()
                            
                        self.begeniy = self.lineEdit_4.text()
                        
                        if self.lineEdit_5.text()==None:
                            self.takips = 2000
                        else:
                            self.takips = self.lineEdit_5.text()
                        session = InstaPy(username=self.usern, password=self.passw,headless_browser=True)
                        session.set_quota_supervisor(enabled=True,peak_follows_daily=int(self.takips))
                        session.login()
                        
                        session.set_do_follow(self.follows, percentage=int(self.followp))
                        session.like_by_tags(self.tagss, amount=int(self.likem))
                        session.end()
                    except ValueError:
                        self.label_7.setText("Hatalı Değer Girdiniz, Lütfen Değerleri Kontrol Ediniz")
                    except:
                        self.label_7.setText("Hata! Lütfen Biraz Bekleyip Tekrar Deneyiniz.")
