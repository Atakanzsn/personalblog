Taş- Kağıt- Makas

Bilgisayara karşı şansınız sınırlarını zorla, bakalım neler yapabileeksin!

Kullanımı:

Önce sen seç, bilgisayarın seçimini göreceksin. Bakalım kazanabilecek misin?

Kodlar:

            import random
            import tkinter
            stats = []

            def get_winner(call):
                if random.random() <= (1 / 3):
                    throw = "Taş"
                elif (1 / 3) <= random.random() <= (2 / 3):
                    throw = "Makas"
                else:
                    throw = "Kağıt"

                if (throw == "Taş" and call == "Kağıt") or (throw == "Kağıt" and call == "Makas") or (throw == "Makas" and call == "Taş"):
                    stats.append("W")
                    result = "Kazandın!"
                elif throw == call:
                    stats.append("D")
                    result = "Berabere!"
                else:
                    stats.append("L")
                    result = "Kaybettin!"

                global output
                output.config(text="Bilgisayar Seçimi: " + throw + "\n" + result)

            def pass_s():
                get_winner("Makas")
            def pass_r():
                get_winner("Taş")
            def pass_p():
                get_winner("Kağıt")

            window = tkinter.Tk()

            scissors = tkinter.Button(window, text="Makas", bg="#ff9999", padx=10, pady=5,command=pass_s, width=20)
            rock = tkinter.Button(window, text="Taş", bg="#80ff80", padx=10, pady=5,command=pass_r, width=20)
            paper = tkinter.Button(window, text="Kağıt", bg="#3399ff", padx=10, pady=5,command=pass_p, width=20)
            output = tkinter.Label(window, width=20 ,fg="red", text="Seçimin Ne??")

            scissors.pack(side="left")
            rock.pack(side="left")
            paper.pack(side="left")
            output.pack(side="right")
            window.mainloop()

            print("Sen: ", stats.count("W"),"\nBilgisayar: ",stats.count("L"))
            for i in stats: 
                if stats.count("L") > stats.count("W"):
                    result = "\nHAHA Bir Makineye Kaybettin!."
                elif stats.count("L") == stats.count("W"):
                    result = "\nBerabere!"
                else:
                    result = "\nBu Seferlik Kazandın...."
            print(result)
